염소산칼륨 + 염산 $$\ce{ \underset{ \text{ 염소산칼륨 }}{ 2KClO3 } + 2 \underset{ \text{ 염산 }}{ HCl } → \underset{ \text{ 염화칼륨 }}{ 2KCl } + \underset{ \text{ 이산화염소 }}{ 2ClO2 } + \underset{ \text{ 과산화수소 }}{ H2O2 } }$$
염소산칼륨 + 황산 $$\ce{ \underset{ \text{ 염소산칼륨 }}{ 6KClO3 } + 3 \underset{ \text{ 황산 }}{ H2SO4 } → 3 \underset{ \text{ 황산칼륨 }}{ K2SO4 } + \underset{ \text{ 이산화염소 }}{ 4ClO2 } + 2 \underset{ \text{ 물 }}{ H2O } + \underset{ \text{ 과염소산 }}{ 2HClO4 } }$$
염소산칼륨 400℃ 분해 $$\ce{ \underset{ \text{ 염소산칼륨 }}{ 2KClO3 } → \underset{ \text{ 염화칼륨 }}{ 2KCl } + \underset{ \text{ 산소 }}{ 3O2 } }$$
염소산칼륨 540~560℃ 분해 $$\ce{ \underset{ \text{ 염소산칼륨 }}{ 2KClO3 } → \underset{ \text{ 염화칼륨 }}{ KCl } + \underset{ \text{ 과염소산칼륨 }}{ KClO4 } + \underset{ \text{ 산소 }}{ O2 } }$$
염소산나트륨 + 염산 $$\ce{ \underset{ \text{ 염소산나트륨 }}{ 2NaClO3 } + 2 \underset{ \text{ 염산 }}{ HCl } → \underset{ \text{ 염화소듐 }}{ 2NaCl } + 2 \underset{ \text{ 이산화염소 }}{ ClO2 } + \underset{ \text{ 과산화수소 }}{ H2O2 } }$$
염소산나트륨 + 황산 $$\ce{ \underset{ \text{ 염소산나트륨 }}{ 2NaClO3 } + \underset{ \text{ 황산 }}{ H2SO4 } → \underset{ \text{ 황산나트륨 }}{ Na2SO4 } + 2 \underset{ \text{ 이산화염소 }}{ ClO2 } + \underset{ \text{ 과산화수소 }}{ H2O2 } }$$
염소산나트륨 분해 $$\ce{ \underset{ \text{ 염소산나트륨 }}{ 2NaClO3 } → \underset{ \text{ 염화소듐 }}{ 2NaCl } + \underset{ \text{ 산소 }}{ 3O2 } }$$
과염소산칼륨 분해 $$\ce{ \underset{ \text{ 과염소산칼륨 }}{ KClO4 } → \underset{ \text{ 염화칼륨 }}{ KCl } + \underset{ \text{ 산소 }}{ 2O2 } }$$
과염소산나트륨 분해 $$\ce{ \underset{ \text{ 과염소산나트륨 }}{ NaClO4 } → \underset{ \text{ 염화소듐 } }{ NaCl } + \underset{ \text{ 산소 }}{ 2O2 } }$$
과염소산나트륨 + 염화칼륨 $$\ce{ \underset{ \text{ 과염소산나트륨 }}{ NaClO4 } + \underset{ \text{ 염화칼륨 }}{ KCl } → \underset{ \text{ 과염소산칼륨 }}{ KClO4 } + \underset{ \text{ 염화소듐 }}{ NaCl } }$$
과염소산암모늄 130℃ 분해 $$\ce{ \underset{ \text{ 과염소산암모늄 }}{ NH4ClO4 } → \underset{ \text{ 염화암모늄 }}{ NH4Cl } + \underset{ \text{ 산소 }}{ 2O2 } }$$
과염소산암모늄 300℃ 분해 (폭발) $$\ce{ 2 \underset{ \text{ 과염소산암모늄 }}{ NH4ClO4 } → \underset{ \text{ 질소 }}{ N2 } + \underset{ \text{ 염소 }}{ Cl2 } + \underset{ \text{ 산소 }}{ 2O2 } + 4 \underset{ \text{ 물 }}{ H2O } }$$
과산화칼륨 분해 $$\ce{ \underset{ \text{ 과산화칼륨 }}{ 2K2O2 } → \underset{ \text{ 산화칼륨 }}{ 2K2O } + \underset{ \text{ 산소 }}{ O2 } }$$
과산화칼륨 + 물 $$\ce{ \underset{ \text{ 과산화칼륨 }}{ 2K2O2 } + 2 \underset{ \text{ 물 }}{ H2O } → \underset{ \text{ 수산화칼륨 }}{ 4KOH } + \underset{ \text{ 산소 }}{ O2 } + \text{발열} }$$
과산화칼륨 + 이산화탄소 $$\ce{ \underset{ \text{ 과산화칼륨 }}{ 2K2O2 } + 2 \underset{ \text{ 이산화탄소 }}{ CO2 } → \underset{ \text{ 탄산나트륨 }}{ 2Na2CO3 } + \underset{ \text{ 산소 }}{ O2 } }$$
과산화나트륨 + 초산 $$\ce{ \underset{ \text{ 염소산나트륨 }}{ Na2O2 } + \underset{ \text{ 초산 }}{ 2CH3COOH } → \underset{ \text{ 소듐 아세테이트 }}{ 2CH3COONa } + \underset{ \text{ 과산화수소 }}{ H2O2 } }$$
과산화나트륨 + 염산 $$\ce{ \underset{ \text{ 염소산나트륨 }}{ Na2O2 } + 2 \underset{ \text{ 염산 }}{ HCl } → 2 \underset{ \text{ 염화소듐 }}{ NaCl } + \underset{ \text{ 과산화수소 }}{ H2O2 } }$$
과산화나트륨 + 황산 $$\ce{ \underset{ \text{ 염소산나트륨 }}{ Na2O2 } + \underset{ \text{ 황산 }}{ H2SO4 } → \underset{ \text{ 황산나트륨 }}{ Na2SO4 } + \underset{ \text{ 과산화수소 }}{ H2O2 } }$$
과산화나트륨 + 알코올 $$\ce{ \underset{ \text{ 염소산나트륨 }}{ Na2O2 } + \underset{ \text{ 에틸알콜 }}{ 2C2H5OH } → \underset{ \text{ 소듐 에톡시드 }}{ 2C2H5ONa } + \underset{ \text{ 과산화수소 }}{ H2O2 } }$$
과산화칼슘 분해 $$\ce{ 2 \underset{ \text{ 과산화칼슘 }}{ CaO2 } → \underset{ \text{ 산화칼슘 }}{ 2CaO } + \underset{ \text{ 산소 }}{ O2 } }$$
과산화칼슘 + 물 $$\ce{ 2 \underset{ \text{ 과산화칼슘 }}{ CaO2 } + 2 \underset{ \text{ 물 }}{ H2O } → \underset{ \text{ 수산화칼슘 }}{ 2Ca(OH)2 } + \underset{ \text{ 산소 }}{ O2 } + \text{ 발열 } }$$
과산화칼슘 + 염산 $$\ce{ \underset{ \text{ 과산화칼슘 }}{ CaO2 } + 2 \underset{ \text{ 염산 }}{ HCl } → \underset{ \text{ 염화칼슘 }}{ CaCl2 } + \underset{ \text{ 과산화수소 }}{ H2O2 } }$$
과산화바륨 분해 $$\ce{ \underset{ \text{ 과산화바륨 }}{ 2BaO2 } → \underset{ \text{ 산화바륨 }}{ 2BaO } + \underset{ \text{ 산소 }}{ O2 } }$$
과산화바륨 + 물 $$\ce{ 2 \underset{ \text{ 과산화바륨 }}{ BaO2 } + 2 \underset{ \text{ 물 }}{ H2O } → \underset{ \text{ 수산화바륨 }}{ 2Ba(OH)2 } + \underset{ \text{ 산소 }}{ O2 } + \text{ 발열 } }$$
과산화바륨 + 염산 $$\ce{ \underset{ \text{ 과산화바륨 }}{ BaO2 } + 2 \underset{ \text{ 염산 }}{ HCl } → \underset{ \text{ 염화바륨 }}{ BaCl2 } + \underset{ \text{ 과산화수소 }}{ H2O2 } }$$
과산화바륨 + 황산 $$\ce{ \underset{ \text{ 과산화바륨 }}{ BaO2 } + \underset{ \text{ 황산 }}{ H2SO4 } → \underset{ \text{ 황산바륨 }}{ BaSO4 } + \underset{ \text{ 과산화수소 }}{ H2O2 } }$$
과산화마그네슘 분해 $$\ce{ \underset{ \text{ 과산화마그네슘 }}{ 2MgO2 } → \underset{ \text{ 산화마그네슘 }}{ 2MgO } + \underset{ \text{ 산소 }}{ O2 } }$$
과산화마그네슘 + 물 $$\ce{ \underset{ \text{ 과산화마그네슘 }}{ 2MgO2 } + 2 \underset{ \text{ 물 }}{ H2O } → \underset{ \text{ 수산화마그네슘 }}{ 2Mg(OH)2 } + \underset{ \text{ 산소 }}{ O2 } + \text{ 발열 } }$$
과산화마그네슘 + 염산 $$\ce{ \underset{ \text{ 과산화마그네슘 }}{ MgO2 } + 2 \underset{ \text{ 염산 }}{ HCl } → \underset{ \text{ 염화마그네슘 }}{ MgCl2 } + \underset{ \text{ 과산화수소 }}{ H2O2 } }$$
질산칼륨(초석) 분해반응식(400[℃]) $$\ce{ 2 \underset{ \text{ 질산칼륨 }}{ KNO3 } → \underset{ \text{ 아질산칼륨 }}{ 2KNO2 } + \underset{ \text{ 산소 }}{ O2 } }$$
질산나트륨(칠레초석) 분해반응식(380[℃]) $$\ce{ 2 \underset{ \text{ 질산나트륨 }}{ NaNO3 } → \underset{ \text{ 아질산나트륨 }}{ 2NaNO2 } + \underset{ \text{ 산소 }}{ O2 } }$$
질산암모늄 가열 $$\ce{ \underset{ \text{ 질산암모늄 }}{ NH4NO3 } → \underset{ \text{ 아산화질소 }}{ N2O } + 2 \underset{ \text{ 물 }}{ H2O } }$$
질산암모늄 분해 $$\ce{ \underset{ \text{ 질산암모늄 }}{ NH4NO3 } → \underset{ \text{ 질소 }}{ N2 } + \underset{ \text{ 산소 }}{ 1/2 O2 } + 2 \underset{ \text{ 물 }}{ H2O } }$$
질산암모늄의 폭발 $$\ce{ 2 \underset{ \text{ 질산암모늄 }}{ NH4NO3 } → 4 \underset{ \text{ 물 }}{ H2O } + \underset{ \text{ 질소 }}{ 2N2 } + \underset{ \text{ 산소 }}{ O2 } }$$
안포폭약 (질산암모늄 94% + 경유 6%) $$\ce{ 3 \underset{ \text{ 질산암모늄 }}{ NH4NO3 } + \underset{ \text{ 탄화수소 }}{ CH2 } → \underset{ \text{ 질소 }}{ 3N2 } + \underset{ \text{ 이산화탄소 }}{ CO2 } + 7 \underset{ \text{ 물 }}{ H2O } }$$
질산은 분해 $$\ce{ 2 \underset{ \text{ 질산은 }}{ AgNO3 } → \underset{ \text{ 은 }}{ 2Ag } + \underset{ \text{ 아산화질소 }}{ 2NO2 } + \underset{ \text{ 산소 }}{ O2 } }$$
질산은 + 염산 $$\ce{ \underset{ \text{ 질산은 }}{ AgNO3 } + \underset{ \text{ 염산 }}{ HCl } → \underset{ \text{ 염화은 }}{ AgCl } + \underset{ \text{ 질산 }}{ HNO3 } }$$
과망간산칼륨 분해 $$\ce{ \underset{ \text{ 과망간산칼륨 }}{ 2KMnO4 } → \underset{ \text{ 망간산칼륨 }}{ K2MnO4 } + \underset{ \text{ 이산화망간 }}{ MnO2 } + \underset{ \text{ 산소 }}{ O2 } }$$
과망간산칼륨 + 묽은 황산 $$\ce{ \underset{ \text{ 과망간산칼륨 }}{ 4KMnO4 } + 6 \underset{ \text{ 황산 }}{ H2SO4 } → 2 \underset{ \text{ 황산칼륨 }}{ K2SO4 } + \underset{ \text{ 황산망간 }}{ 4MnSO4 } + 6 \underset{ \text{ 물 }}{ H2O } + \underset{ \text{ 산소 }}{ 5O2 } }$$
과망간산칼륨 + 진한 황산 $$\ce{ 2 \underset{ \text{ 과망간산칼륨 }}{ KMnO4 } + \underset{ \text{ 황산 }}{ H2SO4 } → \underset{ \text{ 황산칼륨 }}{ K2SO4 } + \underset{ \text{ 과망간산 }}{ 2HMnO4 } }$$
과망간산칼륨 + 염산 $$\ce{ 4 \underset{ \text{ 과망간산칼륨 }}{ KMnO4 } + 12 \underset{ \text{ 염산 }}{ HCl } → 4 \underset{ \text{ 염화칼륨 }}{ KCl } + 4 \underset{ \text{ 염화망간 }}{ MnCl2 } + 6 \underset{ \text{ 물 }}{ H2O } + \underset{ \text{ 산소 }}{ 5O2 } }$$
과망간산칼륨 + 염산 $$\ce{ 2 \underset{ \text{ 과망간산칼륨 }}{ KMnO4 } + 16 \underset{ \text{ 염산 }}{ HCl } → 2 \underset{ \text{ 염화칼륨 }}{ KCl } + 2 \underset{ \text{ 염화망간 }}{ MnCl2 } + 8 \underset{ \text{ 물 }}{ H2O } + \underset{ \text{ 염소 }}{ 5Cl2 } }$$
중크롬산칼륨 분해 $$\ce{ \underset{ \text{ 중크롬산칼륨 }}{ 4K2Cr2O7 } → 2 \underset{ \text{ 산화크롬 }}{ Cr2O3 } + 4 \underset{ \text{ 크롬산칼륨 }}{ Cr2O3 } + \underset{ \text{ 산소 }}{ 3O2 } }$$
중크롬산암모늄 분해 $$\ce{ \underset{ \text{ 중크롬산암모늄 }}{ (NH4)2Cr2O7 } → \underset{ \text{ 산화크롬 }}{ Cr2O3 } + \underset{ \text{ 질소 }}{ N2 } + 4 \underset{ \text{ 물 }}{ H2O } }$$
삼산화크롬(=무수크롬산) 분해 $$\ce{ \underset{ \text{ 삼산화크롬 }}{ 4CrO3 } → \underset{ \text{ 산화크롬 }}{ 2Cr2O3 } + \underset{ \text{ 산소 }}{ 3O2 } }$$
