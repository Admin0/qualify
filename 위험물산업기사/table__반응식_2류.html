2류 가연성고체 반응식
반응 반응식
삼황화린 연소 $$\ce{ \underset{ \text{ 삼황화린 }}{ P4S3 } + \underset{ \text{ 산소 }}{ 8O2 } → \underset{ \text{ 오산화인 }}{ 2P2O5 } + \underset{ \text{ 이산화황 }}{ 3SO2 } }$$
오황화린 연소 $$\ce{ \underset{ \text{ 오황화린 }}{ 2P2S5 } + \underset{ \text{ 산소 }}{ 15O2 } → \underset{ \text{ 오산화인 }}{ 2P2O5 } + \underset{ \text{ 이산화황 }}{ 10SO2 } }$$
오황화린 + 물 $$\ce{ \underset{ \text{ 오황화린 }}{ P2S5 } +\underset{ \text{ 물 }}{ 8H2O }→ \underset{ \text{ 황화수소 }}{ 5H2S } + \underset{ \text{ 인산 }}{ 2H3PO4 } }$$
적린(붉은인) + 염소산염류 $$\ce{ \underset{ \text{ 인 }}{ 6P } + \underset{ \text{ 염소산포타슘 }}{ 5KClO3 } → \underset{ \text{ 염화포타슘 }}{ 5KCl } + \underset{ \text{ 오산화인 }}{ 3P2O5 } }$$
적린(붉은인) 연소 $$\ce{ \underset{ \text{ 인 }}{ 4P } + \underset{ \text{ 산소 }}{ 5O2 } → \underset{ \text{ 오산화인 }}{ 2P2O5 } }$$
유황(황) 연소 $$\ce{ \underset{ \text{ 황 }}{ S } + \underset{ \text{ 산소 }}{ O2 } → \underset{ \text{ 이산화황 }}{ SO2 } }$$
유황(황) + 물 $$\ce{ \underset{ \text{ 이산화황 }}{ SO2 } + \underset{ \text{ 물 }}{ H2O } → \underset{ \text{ 황산 }}{ H2SO4 } }$$
유황 + 수소 $$\ce{ \underset{ \text{ 황 }}{ S } + \underset{ \text{ 수소 }}{ H2 } → \underset{ \text{ 황화수소 }}{ H2S } + \text{ 발열 } }$$
유황 + 철 $$\ce{ \underset{ \text{ 황 }}{ S } + \underset{ \text{ 철 }}{ Fe } → \underset{ \text{ 황화철 }}{ FeS } + \text{ 발열 } }$$
유황 + 탄소 $$\ce{ \underset{ \text{ 황 }}{ 2S } + \underset{ \text{ 탄소 }}{ C } → \underset{ \text{ 이황화탄소 }}{ CS2 } + \text{ 발열 } }$$
철분 연소 $$\ce{ \underset{ \text{ 철 }}{ 4Fe } + \underset{ \text{ 산소 }}{ 3O2 } → \underset{ \text{ 산화제2철 }}{ 2Fe2O3 } }$$
철분 + 염산 $$\ce{ \underset{ \text{ 철 }}{ Fe } + \underset{ \text{ 염산 }}{ 2HCl } → \underset{ \text{ 염화철 }}{ FeCl2 } + \underset{ \text{ 수소 }}{ H2 } }$$
알루미늄분 연소 $$\ce{\underset{ \text{ 알루미늄 }}{ 4Al } + \underset{ \text{ 산소 }}{ 3O2 } → \underset{ \text{ 산화알루미늄 }}{ 2Al2O3 } }$$
알루미늄분 + 염산 $$\ce{\underset{ \text{ 알루미늄 }}{ 2Al } + \underset{ \text{ 염산 }}{ 6HCl } → \underset{ \text{ 염화알루미늄 }}{ 2AlCl3 } +\underset{ \text{ 수소 }}{ 3H2 }}$$
알루미늄분 + 물 $$\ce{\underset{ \text{ 알루미늄 }}{ 2Al } + \underset{ \text{ 물 }}{ 6H2O }→ \underset{ \text{ 수산화알루미늄 }}{ 2Al(OH)3 } +\underset{ \text{ 수소 }}{ 3H2 }}$$
알루미늄분 + 알칼리용액 $$\ce{\underset{ \text{ 알루미늄 }}{ 2Al } + \underset{ \text{ 수산화포타슘 }}{ 2KOH } +\underset{ \text{ 물 }}{ 2H2O }→ \underset{ \text{ 알루미늄산칼륨 }}{ 2KAlO2 } +\underset{ \text{ 수소 }}{ 3H2 }}$$
알루미늄분 + 산화철 $$\ce{\underset{ \text{ 알루미늄 }}{ 8Al } + \underset{ \text{ 산화철 }}{ 3Fe3O4 } → \underset{ \text{ 산화알루미늄 }}{ 4Al2O3 } + \underset{ \text{ 철 }}{ 9Fe } }$$
알루미늄분 테르밋 반응 $$\ce{\underset{ \text{ 알루미늄 }}{ 2Al } + \underset{ \text{ 산화제2철 }}{ Fe2O3 } → \underset{ \text{ 철 }}{ 2Fe } + \underset{ \text{ 산화알루미늄 }}{ Al2O3 } }$$
알루미늄분 + 수산화나트륨수용액 $$\ce{\underset{ \text{ 알루미늄 }}{ 2Al } + \underset{ \text{ 수산화소듐 }}{ 2NaOH } + \underset{ \text{ 물 }}{ 2H2O } → \underset{ \text{ 알루미늄산소듐 }}{ 2NaAlO2 } +\underset{ \text{ 수소 }}{ 3H2 }}$$
아연분 + 물 $$\ce{ \underset{ \text{ 아연 }}{ Zn } +\underset{ \text{ 물 }}{ 2H2O } → \underset{ \text{ 수산화아연 }}{ Zn(OH)2 } + \underset{ \text{ 수소 }}{ H2 } }$$
아연분 + 염산 $$\ce{ \underset{ \text{ 아연 }}{ Zn } + \underset{ \text{ 염산 }}{ 2HCl } → \underset{ \text{ 염화아연 }}{ ZnCl2 } + \underset{ \text{ 수소 }}{ H2 } }$$
아연분 + 황산 $$\ce{ \underset{ \text{ 아연 }}{ Zn } + \underset{ \text{ 황산 }}{ H2SO4 } → \underset{ \text{ 황산아연 }}{ ZnSO4 } + \underset{ \text{ 수소 }}{ H2 } }$$
아연분 + 초산 $$\ce{ \underset{ \text{ 아연 }}{ Zn } + \underset{ \text{ 초산 }}{ 2CH3COOH } → \underset{ \text{ 아세트산아연 }}{ (CH3COO)2Zn } + \underset{ \text{ 수소 }}{ H2 } }$$
마그네슘 + 물 $$\ce{ \underset{ \text{ 마그네슘 }}{ Mg } +\underset{ \text{ 물 }}{ 2H2O }→ \underset{ \text{ 수산화마그네슘 }}{ Mg(OH)2 } + \underset{ \text{ 수소 }}{ H2 } }$$
마그네슘 + 염산 $$\ce{ \underset{ \text{ 마그네슘 }}{ Mg } + \underset{ \text{ 염산 }}{ 2HCl } → \underset{ \text{ 염화마그네슘 }}{ MgCl2 } + \underset{ \text{ 수소 }}{ H2 } }$$
마그네슘 + 황산 $$\ce{ \underset{ \text{ 마그네슘 }}{ Mg } + \underset{ \text{ 황산 }}{ H2SO4 } → \underset{ \text{ 황산마그네슘 }}{ MgSO4 } + \underset{ \text{ 수소 }}{ H2 } }$$
마그네슘 + 이산화탄소 $$\ce{ \underset{ \text{ 마그네슘 }}{ 2Mg } + \underset{ \text{ 이산화탄소 }}{ CO2 } → \underset{ \text{ 산화마그네슘 }}{ 2MgO } + \underset{ \text{ 탄소 }}{ C } }$$
마그네슘 연소 $$\ce{ \underset{ \text{ 마그네슘 }}{ 2Mg } + \underset{ \text{ 산소 }}{ O2 } → \underset{ \text{ 산화마그네슘 }}{ 2MgO } + \text{ 발열 } }$$
