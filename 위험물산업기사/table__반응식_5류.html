5류 자기반응성물질 반응식
반응 반응식
니트로셀룰로오스(NC) 분해 $$\ce{ \underset{ \text{ 니트로셀룰로오스 }}{ 2C24H29O9(ONO2)11 } → \underset{ \text{ 이산화탄소 }}{ 24CO2 } + \underset{ \text{ 일산화탄소 }}{ 24CO } + \underset{ \text{ 질소 }}{ 11N2 } + \underset{ \text{ 수소 }}{ 17H2 } + \underset{ \text{ 물 }}{ 12H2O } }$$
니트로글리세린(NG) 분해 $$\ce{ \underset{ \text{ 니트로글리세린 }}{ 4C3H5(ONO2)3 } → \underset{ \text{ 이산화탄소 }}{ 12CO2 } + \underset{ \text{ 질소 }}{ 6N2 } + \underset{ \text{ 산소 }}{ O2 } + \underset{ \text{ 물 }}{ 10H2O } }$$
질산메틸 제조 $$\ce{ \underset{ \text{ 메탄올 }}{ CH3OH } + \underset{ \text{ 질산 }}{ HNO3 } → \underset{ \text{ 메틸 니트레이드 }}{ CH3ONO2 } + \underset{ \text{ 물 }}{ H2O } }$$
질산에틸 제조 $$\ce{ \underset{ \text{ 에탄올 }}{ C2H5OH } + \underset{ \text{ 질산 }}{ HNO3 } → \underset{ \text{ 에틸 니트레이드 }}{ C2H5ONO2 } + \underset{ \text{ 물 }}{ H2O } }$$
트리니트로톨루엔(TNT) 분해 $$\ce{ \underset{ \text{ 트리니트로톨루엔 }}{ 2C6H2CH3(NO2)3 } → \underset{ \text{ 탄소 }}{ 2C } + \underset{ \text{ 질소 }}{ 3N2 } + \underset{ \text{ 수소 }}{ 5H2 } + \underset{ \text{ 일산화탄소 }}{ 12CO } }$$
트리니트로톨루엔 원료 $$\ce{ \underset{ \text{ 톨루엔 }}{ C6H5CH3 } , \underset{ \text{ 황산 }}{ H2SO4 } , \underset{ \text{ 질산 }}{ HNO3 } }$$
트리니트로페놀(=피크르산) 분해 $$\ce{ \underset{ \text{ 트리니트로페놀 }}{ 2C6H2(OH)(NO2)3 } → \underset{ \text{ 이산화탄소 }}{ 4CO2 } + \underset{ \text{ 일산화탄소 }}{ 6CO } + \underset{ \text{ 탄소 }}{ 2C } + \underset{ \text{ 질소 }}{ 3N2 } + \underset{ \text{ 수소 }}{ 3H2 } }$$
