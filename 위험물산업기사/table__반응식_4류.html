4류 인화성액체 반응식
반응 반응식
디에틸에테르(에테르) 연소 $$\ce{ \underset{ \text{ 디에틸에테르 }}{ C2H5OC2H5 } + \underset{ \text{ 산소 }}{ 6O2 } → \underset{ \text{ 이산화탄소 }}{ 4CO2 } + \underset{ \text{ 물 }}{ 5H2O } }$$
이황화탄소 연소 $$\ce{ \underset{ \text{ 이황화탄소 }}{ CS2 } + \underset{ \text{ 산소 }}{ 3O2 } → \underset{ \text{ 이산화탄소 }}{ CO2 } + \underset{ \text{ 이산화황 }}{ 2SO2 } }$$
이황화탄소 물 (150℃) $$\ce{ \underset{ \text{ 이황화탄소 }}{ CS2 } + \underset{ \text{ 물 }}{ 2H2O } → \underset{ \text{ 이산화탄소 }}{ CO2 } + \underset{ \text{ 황화수소 }}{ 2H2S } + \text{파란불꽃} }$$
아세트알데히드 연소 $$\ce{ 2 \underset{ \text{ 아세트알데히드 }}{ CH3CHO } + \underset{ \text{ 산소 }}{ 5O2 } → \underset{ \text{ 이산화탄소 }}{ 4CO2 } + \underset{ \text{ 물 }}{ 4H2O } }$$
아세트알데히드 산화 $$\ce{ \underset{ \text{ 아세트알데히드 }}{ CH3CHO } + \underset{ \text{ 산소 }}{ 1/2 O2 } → \underset{ \text{ 아세트산 }}{ CH3COOH } }$$
산화프로필렌 연소 $$\ce{ CH3CHCH2O + \underset{ \text{ 산소 }}{ 4O2 } → \underset{ \text{ 이산화탄소 }}{ 3CO2 } + \underset{ \text{ 물 }}{ 3H2O } }$$
벤젠 연소 $$\ce{ \underset{ \text{ 벤젠 }}{ 2C6H6 } + \underset{ \text{ 산소 }}{ 15O2 } → \underset{ \text{ 이산화탄소 }}{ 12CO2 } + \underset{ \text{ 물 }}{ 6H2O } }$$
벤젠 + 황산 $$\ce{ \underset{ \text{ 벤젠 }}{ C6H6 } + \underset{ \text{ 황산 }}{ 2H2SO4 } → \underset{ \text{ 벤젠술폰산 }}{ C6H5SO3H } + \underset{ \text{ 물 }}{ H2O } }$$
톨루엔 연소 $$\ce{ \underset{ \text{ 톨루엔 }}{ C6H5CH3 } + \underset{ \text{ 산소 }}{ 9O2 } → \underset{ \text{ 이산화탄소 }}{ 7CO2 } + \underset{ \text{ 물 }}{ 4H2O } }$$
아세톤 연소 $$\ce{ \underset{ \text{ 아세톤 }}{ CH3COCH3 } + \underset{ \text{ 산소 }}{ 4O2 } → \underset{ \text{ 이산화탄소 }}{ 3CO2 } + \underset{ \text{ 물 }}{ 3H2O } }$$
메틸에틸케톤 완전연소 $$\ce{ \underset{ \text{ 메틸에틸케톤 }}{ 2CH3COC2H5 } + \underset{ \text{ 산소 }}{ 11O2 } → \underset{ \text{ 이산화탄소 }}{ 8CO2 } + \underset{ \text{ 물 }}{ 8H2O } }$$
메틸알코올 연소 $$\ce{ 2 \underset{ \text{ 메탄올 }}{ CH3OH } + \underset{ \text{ 산소 }}{ 3O2 } → \underset{ \text{ 이산화탄소 }}{ 2CO2 } + \underset{ \text{ 물 }}{ 4H2O } }$$
메틸알코올 + 알칼리금속 $$\ce{ \underset{ \text{ 소듐 }}{ 2Na } + \underset{ \text{ 메탄올 }}{ 2CH3OH } → 2CH3ONa + \underset{ \text{ 수소 }}{ H2 } }$$
메틸알코올 산화 $$\ce{ \underset{ \text{ 메탄올 }}{ CH3OH } → \underset{ \text{ 포름알데히드 }}{ HCHO } → \underset{ \text{ 포름산 }}{ HCOOH } }$$
메틸알코올 환원 $$\ce{ \underset{ \text{ 메탄올 }}{ CH3OH } ← \underset{ \text{ 포름알데히드 }}{ HCHO } ← \underset{ \text{ 포름산 }}{ HCOOH } }$$
에틸알코올 + 칼륨 $$\ce{ 2 \underset{ \text{ 에탄올 }}{ C2H5OH } + \underset{ \text{ 칼륨 }}{ 2K } → \underset{ \text{ 칼륨에틸레이트 }}{ 2C2H5OK } + \underset{ \text{ 수소 }}{ H2 } }$$
에틸알코올 연소 $$\ce{ \underset{ \text{ 에탄올 }}{ C2H5OH } + \underset{ \text{ 산소 }}{ 3O2 } → \underset{ \text{ 이산화탄소 }}{ 2CO2 } + \underset{ \text{ 물 }}{ 3H2O } }$$
에틸알코올 + 알칼리금속 $$\ce{ 2 \underset{ \text{ 에탄올 }}{ C2H5OH } + \underset{ \text{ 소듐 }}{ 2Na } → 2C2H5ONa + \underset{ \text{ 수소 }}{ H2 } }$$
에틸알코올 산화 $$\ce{ \underset{ \text{ 에탄올 }}{ C2H5OH } → \underset{ \text{ 아세트알데히드 }}{ CH3CHO } → \underset{ \text{ 아세트산 }}{ CH3COOH } }$$
에틸알코올 환원 $$\ce{ \underset{ \text{ 에탄올 }}{ C2H5OH } ← \underset{ \text{ 아세트알데히드 }}{ CH3CHO } ← \underset{ \text{ 아세트산 }}{ CH3COOH } }$$
초산(아세트산) 연소 $$\ce{ \underset{ \text{ 아세트산 }}{ CH3COOH } + \underset{ \text{ 산소 }}{ O2 } → \underset{ \text{ 이산화탄소 }}{ 2CO2 } + \underset{ \text{ 물 }}{ 2H2O } }$$
초산(아세트산) + 과산화나트륨 $$\ce{ 2 \underset{ \text{ 아세트산 }}{ CH3COOH } + \underset{ \text{ 과산화나트륨 }}{ Na2O2 } → \underset{ \text{ 아세트산나트륨 }}{ 2CH3COONa } + \underset{ \text{ 과산화수소 }}{ H2O2 } }$$
초산메틸(아세트산메틸) 가수분해 $$\ce{ \underset{ \text{ 초산메틸 }}{ CH3COOCH3 } + \underset{ \text{ 물 }}{ H2O } → \underset{ \text{ 아세트산 }}{ CH3COOH } + \underset{ \text{ 메탄올 }}{ CH3OH } }$$
의산(포름산,개미산) 연소 $$\ce{ 2 \underset{ \text{ 포름산 }}{ HCOOH } + \underset{ \text{ 산소 }}{ O2 } → \underset{ \text{ 이산화탄소 }}{ 2CO2 } + \underset{ \text{ 물 }}{ 2H2O } }$$
의산에틸(개미산에틸) 가수분해 $$\ce{ \underset{ \text{ 개미산에틸 }}{ HCOOC2H5 } + \underset{ \text{ 물 }}{ H2O } → \underset{ \text{ 포름산 }}{ HCOOH } + \underset{ \text{ 에탄올 }}{ C2H5OH } }$$
히드라진 열분해(180[℃]) $$\ce{ \underset{ \text{ 히드라진 }}{ 2N2H4 } → \underset{ \text{ 암모니아 }}{ 2NH3 } + \underset{ \text{ 질소 }}{ N2 } + \underset{ \text{ 수소 }}{ H2 } }$$
에틸렌글리콜 연소 $$\ce{ \underset{ \text{ 에틸렌글리콜 }}{ 2C2H6O2 } + \underset{ \text{ 산소 }}{ 5O2 } → \underset{ \text{ 이산화탄소 }}{ 4CO2 } + \underset{ \text{ 물 }}{ 6H2O } }$$
에틸렌글리콜 연소 $$\ce{ \underset{ \text{ 에틸렌글리콜 }}{ 2CH2OHCH2OH } + \underset{ \text{ 산소 }}{ 5O2 } → \underset{ \text{ 이산화탄소 }}{ 4CO2 } + \underset{ \text{ 물 }}{ 6H2O } }$$
글리세린 연소 $$\ce{ \underset{ \text{ 글리세린 }}{ 2C3H8O3 } + \underset{ \text{ 산소 }}{ 7O2 } → \underset{ \text{ 이산화탄소 }}{ 6CO2 } + \underset{ \text{ 물 }}{ 8H2O } }$$
글리세린 연소 $$\ce{ \underset{ \text{ 글리세린 }}{ 2CH2OHCHOHCH2OH } + \underset{ \text{ 산소 }}{ 7O2 } → \underset{ \text{ 이산화탄소 }}{ 6CO2 } + \underset{ \text{ 물 }}{ 8H2O } }$$
