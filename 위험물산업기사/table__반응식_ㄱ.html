반응 반응식
과망간산칼륨 + 염산 $$\ce{ 2\underset{\text{과망간산칼륨}}{KMnO4} + 16 \underset{ \text{ 염산 }}{ HCl } → 2 \underset{ \text{ 염화칼륨 }}{ KCl } + 2MnCl2 + 8 \underset{ \text{ 물 }}{ H2O } + 5Cl2 }$$
과망간산칼륨 + 황산 $$\ce{ 4\underset{\text{과망간산칼륨}}{KMnO4} + 6 \underset{ \text{ 황산 }}{ H2SO4 } → 2 \underset{ \text{ 황산칼륨 }}{ K2SO4 } + 4 \underset{ \text{ 황산망간 }}{ MnSO4 } + 6 \underset{ \text{ 물 }}{ H2O } + 5 \underset{ \text{ 산소 }}{ O2 } }$$
과망간산칼륨 열분해식 $$\ce{ \underset{\text{과망간산칼륨}}{2KMnO4} → \underset{\text{ 망간산칼륨 }}{ K2MnO4 } + \underset{\text{ 이산화망간 }}{ MnO2 } + \underset{ \text{ 산소 }}{ O2 } }$$
과산화나트륨 + 물 $$\ce{ \underset{\text{과산화나트륨}}{2Na2O2} + 2 \underset{ \text{ 물 }}{ H2O } → 4 \underset{ \text{ 수산화나트륨 }}{ NaOH }+ \underset{\text{ 산소 }}{ O2 } }$$
과산화나트륨 + 이산화탄소 $$\ce{ 2 \underset{ \text{ 과산화나트륨 }}{ Na2O2 } + 2 \underset{\text{이산화탄소}}{ CO2 } → 2 \underset{ \text{ 탄산나트륨 }}{ Na2CO3 } + \underset{\text{산소}}{O2} }$$
과산화바륨 + 물 $$\ce{ \underset{ \text{ 과산화바륨 }}{ 2BaO2 } + 2 \underset{ \text{ 물 }}{ H2O } + 2 \underset{ \text{ 수산화바륨 }}{ Ba(OH)2 } → \underset{\text{산소}}{O2} }$$
과산화바륨 + 염산 $$\ce{ \underset{ \text{ 과산화바륨 }}{ BaO2 } + 2 \underset{ \text{ 염산 }}{ HCl } → \underset{ \text{ 염화바륨 }}{ BaCl2 }+ \underset{ \text{ 과산화수소 }}{ H2O2 }}$$
과산화바륨 + 황산 $$\ce{ \underset{ \text{ 과산화바륨 }}{ BaO2 } + \underset{ \text{ 황산 }}{ H2SO4 } → \underset{ \text{ 황산바륨 }}{ BaSO4 }+ \underset{ \text{ 과산화수소 }}{ H2O2 } }$$
과산화수소 + 히드라진 $$\ce{2 \underset{ \text{ 과산화수소 }}{ H2O2 } + \underset{ \text{ 히드라진 }}{ N2H4 } → 4 \underset{ \text{ 물 }}{ H2O } + \underset{ \text{ 질소 }}{ N2 } }$$
과산화수소 분해식 $$\ce{2 \underset{ \text{ 과산화수소 }}{ H2O2 } → 2 \underset{ \text{ 물 }}{ H2O } + \underset{\text{산소}}{O2} }$$
과산화칼륨 + 물 $$\ce{ \underset{ \text{ 과산화칼륨 }}{ 2K2O2 }+ 2 \underset{ \text{ 물 }}{ H2O } → 4 \underset{ \text{ 수산화칼륨 }}{ KOH }+ \underset{\text{산소}}{O2} }$$
과염소산 분해식 $$\ce{ \underset{ \text{ 과염소산 }}{ HClO4 } → \underset{ \text{ 염산 }}{ HCl } + 2 \underset{ \text{ 산소 }}{ O2 } }$$
과염소산나트륨 + 염화칼륨 $$\ce{ \underset{ \text{ 과염소산나트륨 }}{ NaClO4 } + \underset{ \text{ 염화칼륨 }}{ KCl } → \underset{ \text{ 과염소산칼륨 }}{ KClO4 } + \underset{ \text{ 염화나트륨 }}{ NaCl } }$$
과염소산칼륨 610℃ 열분해식 $$\ce{ \underset{ \text{ 과염소산칼륨 }}{ KClO4 } → \underset{ \text{ 염화칼륨 }}{ KCl } + 2 \underset{\text{산소}}{ O2 } }$$
구리 + 진한질산 $$\ce{ \underset{ \text{ 구리 }}{ Cu } + \underset{ \text{ 진한질산 }}{ 4HNO3 } → \underset{ \text{ 질산구리 }}{ Cu(NO3)2 } + \underset{ \text{ 이산화질소 }}{ 2NO2 } + 2 \underset{ \text{ 물 }}{ H2O } }$$
